Sub СопоставитьПризнаки()

Dim ws As Worksheet
Dim i As Long, j As Long, LastRowA As Long, LastRowI As Long

Set ws = ThisWorkbook.Sheets("Sheet1") 'Замените "Sheet1" на имя вашего листа

'Определение последней строки в столбцах A и I
LastRowA = ws.Range("A" & Rows.Count).End(xlUp).Row
LastRowI = ws.Range("I" & Rows.Count).End(xlUp).Row

'Перебор всех ID в столбце A
For i = 2 To LastRowA
    'Поиск совпадения ID в столбце I
    For j = 2 To LastRowI
        If ws.Range("A" & i).Value = ws.Range("I" & j).Value Then
            'Запись соответствующего признака в столбец G
            ws.Range("G" & i).Value = ws.Range("J" & j - 1).Value
            Exit For
        End If
    Next j
Next i

End Sub
